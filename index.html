// Вставьте этот код в Apps Script таблицы: // Расширения → Apps Script → Новый проект function doGet(e) { const action = e.parameter.action; const ss = SpreadsheetApp.openById(e.parameter.id || 'YOUR_ID'); if (action === 'read') { const sh = ss.getSheetByName(e.parameter.sheet); const data = sh.getDataRange().getValues(); return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } if (action === 'append') { const sh = ss.getSheetByName(e.parameter.sheet); const row = JSON.parse(e.parameter.row); sh.appendRow(row); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } } // После публикации: Развернуть → Новое развертывание // Тип: Веб-приложение, Доступ: Все
